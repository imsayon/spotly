function useToasts(){
  const [toasts,setToasts]=useState([]);
  const add=useCallback((msg,type='info')=>{
    const id=Date.now();
    setToasts(p=>[...p,{id,msg,type}]);
    setTimeout(()=>setToasts(p=>p.filter(t=>t.id!==id)),3200);
  },[]);
  return{toasts,add};
}

function CountUp({target,duration=1500,suffix='',prefix='',style={}}){
  const [val,setVal]=useState(0);
  const ref=useRef(null);
  useEffect(()=>{
    const start=Date.now();
    const isFloat=String(target).includes('.');
    const t=setInterval(()=>{
      const elapsed=Date.now()-start;
      const p=Math.min(elapsed/duration,1);
      const ease=1-Math.pow(1-p,3);
      setVal(isFloat?(ease*target).toFixed(1):Math.round(ease*target));
      if(p>=1)clearInterval(t);
    },16);
    return()=>clearInterval(t);
  },[target,duration]);
  return <span style={style}>{prefix}{val}{suffix}</span>;
}

function ConsumerQueue({entry,onLeave,onServed,addToast}){
  const [status,setStatus]=useState('WAITING');
  const [position,setPosition]=useState(2);
  const [currentToken,setCurrentToken]=useState(41);
  const [seconds,setSeconds]=useState(position*240);
  const [queueEntries,setQueueEntries]=useState(QUEUE_ENTRIES);

  useEffect(()=>{
    const t=setInterval(()=>{
      setPosition(p=>{
        if(p<=0){clearInterval(t);setStatus('CALLED');addToast("It's your turn! 🔔",'success');return 0;}
        setCurrentToken(ct=>ct+1);
        setSeconds(s=>Math.max(0,s-240));
        return p-1;
      });
    },5000);
    return()=>clearInterval(t);
  },[]);

  // Countdown timer
  useEffect(()=>{
    if(status!=='WAITING') return;
    const t=setInterval(()=>setSeconds(s=>Math.max(0,s-1)),1000);
    return()=>clearInterval(t);
  },[status]);

  const mins=Math.floor(seconds/60);
  const secs=seconds%60;

  const statusStyle={
    WAITING:{color:'#f5c418',bg:'rgba(245,196,24,.07)',border:'rgba(245,196,24,.25)',icon:'⏳',anim:'tokenY'},
    CALLED:{color:'#1fd97c',bg:'rgba(31,217,124,.1)',border:'rgba(31,217,124,.3)',icon:'🔔',anim:'tokenG'},
  };
  const st=statusStyle[status]||statusStyle.WAITING;

  return(
    <div style={{padding:'20px 18px'}}>
      <div style={{display:'flex',alignItems:'center',justifyContent:'space-between',marginBottom:24}}>
        <div>
          <h1 style={{fontFamily:'Syne,sans-serif',fontSize:22,fontWeight:900,marginBottom:2}}>Your Queue</h1>
          <p style={{color:'var(--t3)',fontSize:13}}>{entry.outlet}</p>
        </div>
        <span style={{...s.badge(status==='CALLED'?'green':'yellow')}}>{status==='CALLED'?'🔔 CALLED':'⏳ WAITING'}</span>
      </div>

      {/* TOKEN CIRCLE */}
      <div style={{textAlign:'center',marginBottom:28,position:'relative'}}>
        {status==='CALLED'&&[1,2,3].map(i=>(
          <div key={i} style={{position:'absolute',top:'50%',left:'50%',transform:'translate(-50%,-50%)',width:200+i*40,height:200+i*40,borderRadius:'50%',border:`2px solid rgba(31,217,124,${0.3/i})`,animation:`ripple ${1.8+i*.4}s ease-out ${i*.4}s infinite`,pointerEvents:'none'}}/>
        ))}
        <div style={{width:200,height:200,borderRadius:'50%',margin:'0 auto',background:st.bg,border:`2px solid ${st.border}`,display:'flex',flexDirection:'column',alignItems:'center',justifyContent:'center',animation:`${st.anim} 2.5s infinite`,transition:'all .6s',position:'relative',zIndex:2}}>

          <div style={{fontFamily:'JetBrains Mono,monospace',fontSize:80,fontWeight:800,color:st.color,lineHeight:1,animation:'countBounce .5s ease'}}>{entry.token}</div>
          <div style={{fontSize:12,fontWeight:700,color:st.color}}>{st.icon} {status==='CALLED'?"YOUR TURN!":"Waiting"}</div>
        </div>
      </div>

      {status==='CALLED'?(
        <div className="as" style={{...s.card,background:'rgba(31,217,124,.08)',borderColor:'rgba(31,217,124,.28)',marginBottom:18,padding:'20px',textAlign:'center'}}>
          <div style={{fontSize:40,marginBottom:8}}>🎉</div>

      ):(
        <>
          {/* COUNTDOWN */}
          <div style={{display:'grid',gridTemplateColumns:'1fr 1fr 1fr',gap:12,marginBottom:18}}>
            {[
              {label:'People Ahead',val:<span style={{fontFamily:'JetBrains Mono,monospace',fontSize:36,fontWeight:800,...s.gradCText}}>{position}</span>},

// MERCHANT QUEUE OPERATOR
// ============================================================
function MerchantQueue({addToast}){
  const [entries,setEntries]=useState(QUEUE_ENTRIES);
  const [called,setCalled]=useState(null);
  const [selectedOutlet,setSelectedOutlet]=useState('o1');
  const [calling,setCalling]=useState(false);
  const waiting=entries.filter(e=>e.status==='WAITING');

  const callNext=()=>{
    if(!waiting.length) return;
    setCalling(true);
    setTimeout(()=>{
      const next=waiting[0];
      setEntries(p=>p.map(e=>e.id===next.id?{...e,status:'CALLED'}:e));
      setCalled(next);
      setCalling(false);
      addToast(`Called token #${next.token}`,'success');
    },1100);
  };

  const markServed=()=>{
    if(!called) return;
    setEntries(p=>p.filter(e=>e.id!==called.id));
    addToast(`Token #${called.token} served ✓`,'success');
    setCalled(null);
  };

  const missToken=(e)=>{
    setEntries(p=>p.filter(en=>en.id!==e.id));
    if(called?.id===e.id) setCalled(null);
    addToast(`Token #${e.token} marked missed`,'error');
  };

  return(
    <div style={{padding:'36px 36px 40px'}}>
      <div style={{display:'flex',justifyContent:'space-between',alignItems:'center',marginBottom:26}}>
        <div>
          <h1 style={{fontFamily:'Syne,sans-serif',fontSize:28,fontWeight:900,marginBottom:4,display:'flex',alignItems:'center',gap:8}}>Queue Operator <Ic.Shield/></h1>
          <p style={{color:'var(--t3)',fontSize:14}}>Real-time queue control</p>
        </div>
        <select style={{...s.input,width:'auto',padding:'10px 14px',fontSize:13}} value={selectedOutlet} onChange={e=>setSelectedOutlet(e.target.value)}>
          {OUTLETS.map(o=><option key={o.id} value={o.id}>{o.name}</option>)}
        </select>
      </div>

      <div style={{display:'grid',gridTemplateColumns:'1fr 1fr',gap:16,marginBottom:20}}>
        {/* CURRENTLY SERVING */}
        <div style={{...s.card,padding:'32px',textAlign:'center',background:'rgba(31,217,124,.06)',borderColor:'rgba(31,217,124,.2)',position:'relative',overflow:'hidden'}}>
          <div style={{position:'absolute',top:-30,right:-30,width:110,height:110,borderRadius:'50%',background:'rgba(31,217,124,.08)',filter:'blur(28px)'}}/>
          <div style={{fontSize:11,fontWeight:700,color:'#1fd97c',textTransform:'uppercase',letterSpacing:1.5,marginBottom:10,display:'flex',alignItems:'center',justifyContent:'center',gap:7}}><Ic.Sparkle/>Now Serving</div>
          <div style={{fontFamily:'JetBrains Mono,monospace',fontSize:88,fontWeight:800,color:'#fff',lineHeight:1,marginBottom:16,animation:called?'countBounce .5s ease':''}}>{called?`#${called.token}`:'—'}</div>
          {called?(
            <div style={{display:'flex',gap:10,justifyContent:'center'}}>
              <button style={{...s.btnM,gap:6}} onClick={markServed}><Ic.Check/>Served</button>
              <button style={{...s.btnDanger,gap:6}} onClick={()=>missToken(called)}>Miss</button>
            </div>
          ):<p style={{color:'var(--t4)',fontSize:13}}>Call next person to serve</p>}
        </div>

        {/* CALL NEXT */}
        <div style={{...s.card,padding:'32px',textAlign:'center'}}>
          <div style={{fontFamily:'JetBrains Mono,monospace',fontSize:52,fontWeight:800,color:'#fff',marginBottom:4}}>{waiting.length}</div>
          <div style={{fontSize:13,color:'var(--t3)',marginBottom:24}}>customers waiting</div>
          <button style={{...s.btnM,justifyContent:'center',width:'100%',padding:'17px',fontSize:16,borderRadius:14,gap:10,opacity:waiting.length===0?.4:1,cursor:waiting.length===0?'not-allowed':'pointer'}}
            onClick={callNext}
            onMouseEnter={e=>{if(waiting.length>0){e.currentTarget.style.transform='translateY(-3px)';e.currentTarget.style.boxShadow='0 18px 45px rgba(31,217,124,.38)'}}}
            onMouseLeave={e=>{e.currentTarget.style.transform='';e.currentTarget.style.boxShadow=''}}>
            {calling?<div style={{width:20,height:20,border:'2px solid rgba(255,255,255,.3)',borderTop:'2px solid #fff',borderRadius:'50%',animation:'spin .7s linear infinite'}}/>:<><Ic.Bell/>{waiting.length>0?'Call Next Person':'Queue Empty'}</>}
          </button>
          {waiting.length>0&&<p style={{fontSize:11,color:'var(--t4)',marginTop:14}}>Next: Token #{waiting[0]?.token} · {waiting[0]?.initials}</p>}
        </div>
      </div>

      {/* LIVE BOARD */}
      <div style={{...s.card,padding:'22px'}}>
        <div style={{display:'flex',justifyContent:'space-between',alignItems:'center',marginBottom:18}}>
          <h3 style={{fontFamily:'Syne,sans-serif',fontWeight:800,fontSize:16}}>Live Tracking Board</h3>
          <span style={{...s.badge('gray')}}>{entries.length} Total</span>
        </div>
        {entries.length===0?(
          <div style={{textAlign:'center',padding:'36px',color:'var(--t3)'}}>
            <div style={{fontSize:36,marginBottom:10}}>✓</div>
            <p style={{fontWeight:700}}>Queue is clear</p>
          </div>
        ):(
          <div style={{display:'flex',flexDirection:'column',gap:8}}>
            {entries.map(e=>(
              <div key={e.id} style={{display:'flex',alignItems:'center',justifyContent:'space-between',padding:'13px 16px',borderRadius:14,background:e.status==='CALLED'?'rgba(31,217,124,.07)':'rgba(255,255,255,.025)',border:e.status==='CALLED'?'1px solid rgba(31,217,124,.28)':'1px solid var(--bdr)',transition:'all .3s',animation:e.status==='CALLED'?'tokenG 2.5s infinite':''}}>
                <div style={{display:'flex',alignItems:'center',gap:14}}>
                  <div style={{width:42,height:42,borderRadius:11,background:'rgba(255,255,255,.06)',display:'flex',alignItems:'center',justifyContent:'center',fontFamily:'JetBrains Mono,monospace',fontWeight:900,fontSize:16}}>#{e.token}</div>
                  <div style={{width:32,height:32,borderRadius:'50%',background:`hsl(${e.token*47%360},55%,55%)`,display:'flex',alignItems:'center',justifyContent:'center',fontSize:11,fontWeight:800,color:'#fff'}}>{e.initials}</div>
                  <span style={s.badge(e.status==='CALLED'?'green':'yellow')}>{e.status}</span>
                  <span style={{fontSize:12,color:'var(--t3)'}}>{e.wait}</span>
                </div>
                <div style={{display:'flex',gap:8}}>
                  {e.status==='CALLED'&&<button style={{...s.btnGhost,padding:'6px 12px',fontSize:11,gap:5}} onClick={markServed}><Ic.Check/>Served</button>}
                  {e.status==='WAITING'&&<button style={{...s.btnDanger,padding:'6px 12px',fontSize:11}} onClick={()=>missToken(e)}>Miss</button>}
                </div>
              </div>
            ))}
          </div>
        )}
      </div>
    </div>
  );
}
